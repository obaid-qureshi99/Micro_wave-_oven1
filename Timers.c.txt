void timer0_init()
{
//mode of timer operation and prescaler

//normal mode with no prescaler 256
TCCR0=(1<<CS02);


// initialize timer counter to

//enable timer0 overflow interrup
TIMSK|=(1<<TOIE0); //Enalbe overflow interrup
sei();             //Enalbe global interrupt


}
